def get_fitted_values(df, x, y):
    y = df.loc[:,y]
    x = df.loc[:,x]
    x = sm.add_constant(x)
    ols = sm.OLS(y, x).fit()
    return ols.fittedvalues

def test_diff_r(r1, r2, n1, n2):
    z1 = np.arctanh(r1)
    z2 = np.arctanh(r2)
    sd = np.sqrt((1 / (n1 - 3)) + (1 / (n2 - 3)))
    z = np.abs(z1 - z2) / sd
    p_one = stats.norm.sf(z)
    p_two = 2 * p_one
    return z, p_one, p_two 

result = test_diff_r(0.135, 0.137, 1195, 396)
